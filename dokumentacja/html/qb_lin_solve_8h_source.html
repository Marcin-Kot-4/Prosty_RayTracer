8/* *************************************************************************************************
9
10 qbLinSolve
11
12 Function to solve a system of linear equations in the form of y = X*beta, where we
13 want to solve for beta.
14
15 *** INPUTS ***
16
17 aMatrix qbMatrix2<T> The matrix of independent variables (X in the above equation).
18 bVector qbVector<T> The vector of dependent variables (y in the above equation).
19 resultVec qbVector<T> The vector of unknown parameters (beta in the above equation).
20 The final solution is returned in this vector.
21
22 *** OUTPUTS ***
23
24 INT Flag indicating success or failure of the process.
25 1 Indicates success.
26 -1 indicates failure due to there being no unique solution (infinite solutions).
27 -2 indicates failure due to there being no solution.
28
29 Uses Gaussian elimination on the augmented matrix, followed by back substitution.
30
31 ************************************************************************************************* */
32

46// The qbLinSolve function.
47template <typename T>
48int qbLinSolve(const qbMatrix2<T> &aMatrix, const qbVector<T> &bVector, qbVector<T> &resultVec)
49{
50 // Make a copy of the input matrix, aMatrix.
51 // We will use this to create the augmented matrix, so we have
52 // to make a copy.
53 qbMatrix2<T> inputMatrix = aMatrix;
54
55 // Compute the rank of the original matrix.
56 int originalRank = inputMatrix.Rank();
57
58 /* Combine inputMatrix and bVector together into a single matrix,
59 ready for using Gaussian elimination to reduce to
60 row-echelon form. */
61
62 // Extract data from bVector.
63 int numDims = bVector.GetNumDims();
64 std::vector<T> bVecData;
65 for (int i=0; i<numDims; ++i)
66 bVecData.push_back(bVector.GetElement(i));
67
68 // Use this to create a qbMatrix2 object with the same data (nx1).
69 qbMatrix2<T> bMatrix(numDims, 1, bVecData);
70
71 // Combine the two matrices together.
72 inputMatrix.Join(bMatrix);
73
74 /* Use Gaussian elmination to convert to row-echelon form. */
75 qbMatrix2<T> rowEchelonMatrix = inputMatrix.RowEchelon();
76
77 /* Comute the rank of the augmented matrix.
78 Note that we do this after performing Gaussian elimination to
79 reduce the matrix to row echelon form so that if this was
80 successful, there is no need to repeat this operation twice. */
81 int augmentedRank = rowEchelonMatrix.Rank();
82
83 /* *********************************************************************
84 Test the two ranks to determine the nature of the system we
85 are dealing with. The conditions are as follows:
86
87 n = number of rows.
88
89 1) originalRank = augmentedRank = n => A unique solution exists.
90 2) originalRank = augmentedRank < n => An infinite number of solutions exist.
91 3) originalRank < augmentedRank => No solutions exist.
92 ********************************************************************* */
93 if ((originalRank == augmentedRank) && (originalRank < inputMatrix.GetNumRows()))
94 {
95 return QBLINSOLVE_NOUNIQUESOLUTION;
96 }
97 else if (originalRank < augmentedRank)
98 {
99 return QBLINSOLVE_NOSOLUTIONS;
100 }
101 else
102 {
103 /* Create a qbVector object to store the output. Initially we will
104 populate this with the data from bVecData, but we are going to modify
105 the elements as we compute them. */
106 qbVector<T> output(bVecData);
107
108 // Now use back-substitution to compute the result.
109 int numRows = rowEchelonMatrix.GetNumRows();
110 int numCols = rowEchelonMatrix.GetNumCols();
111 int startRow = numRows-1;
112
113 // Loop over the rows, in reverse order.
114 for (int i=startRow; i>=0; --i)
115 {
116 // Extract the currentResult for this row.
117 T currentResult = rowEchelonMatrix.GetElement(i, numCols-1);
118
119 // Compute the cumulative sum.
120 T cumulativeSum = static_cast<T>(0.0);
121 for (int j=i+1; j<numRows; ++j)
122 {
123 cumulativeSum += (rowEchelonMatrix.GetElement(i,j) * output.GetElement(j));
124 }
125
126 // Compute the answer.
127 T finalAnswer = (currentResult - cumulativeSum) / rowEchelonMatrix.GetElement(i,i);
128
129 // And store.
130 output.SetElement(i, finalAnswer);
131
132 }
133
134 // Return the output.
135 resultVec = output;
136
137 }
138
139 return 1;
140}
