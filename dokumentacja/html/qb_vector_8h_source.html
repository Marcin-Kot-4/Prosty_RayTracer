132// Compute the length of the vector,known as the 'norm'.
133template <class T>
134T qbVector<T>::norm()
135{
136 T cumulativeSum = static_cast<T>(0.0);
137 for (int i=0; i<m_nDims; ++i)
138 cumulativeSum += (m_vectorData.at(i) * m_vectorData.at(i));
139
140 return sqrt(cumulativeSum);
141}
142
143// Return a normalized copy of the vector.
144template <class T>
145qbVector<T> qbVector<T>::Normalized()
146{
147 // Compute the vector norm.
148 T vecNorm = this->norm();
149
150 // Compute the normalized version of the vector.
151 qbVector<T> result(m_vectorData);
152 return result * (static_cast<T>(1.0) / vecNorm);
153}
154
155// Normalize the vector in place.
156template <class T>
157void qbVector<T>::Normalize()
158{
159 // Compute the vector norm.
160 T vecNorm = this->norm();
161
162 // Compute the elements of the normalized version of the vector.
163 for (int i=0; i<m_nDims; ++i)
164 {
165 T temp = m_vectorData.at(i) * (static_cast<T>(1.0) / vecNorm);
166 m_vectorData.at(i) = temp;
167 }
168}

248template <class T>
249qbVector<T> qbVector<T>::cross(const qbVector<T> &a, const qbVector<T> &b)
250{
251 // Check that the number of dimensions match.
252 if (a.m_nDims != b.m_nDims)
253 throw std::invalid_argument("Vector dimensions must match for the cross-product to be computed.");
254
255 // Check that the number of dimensions is 3.
256 /* Although the cross-product is also defined for 7 dimensions, we are
257 not going to consider that case at this time. */
258 if (a.m_nDims != 3)
259 throw std::invalid_argument("The cross-product can only be computed for three-dimensional vectors.");
260
261 // Compute the cross product.
262 std::vector<T> resultData;
263 resultData.push_back((a.m_vectorData.at(1) * b.m_vectorData.at(2)) - (a.m_vectorData.at(2) * b.m_vectorData.at(1)));
264 resultData.push_back(-((a.m_vectorData.at(0) * b.m_vectorData.at(2)) - (a.m_vectorData.at(2) * b.m_vectorData.at(0))));
265 resultData.push_back((a.m_vectorData.at(0) * b.m_vectorData.at(1)) - (a.m_vectorData.at(1) * b.m_vectorData.at(0)));
266
267 qbVector<T> result(resultData);
268 return result;
269}
