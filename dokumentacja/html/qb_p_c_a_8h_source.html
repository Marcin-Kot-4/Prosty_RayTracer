26// Function to compute the column means.
27template <typename T>
28std::vector<T> ComputeColumnMeans(const qbMatrix2<T> &inputData)
29{
30 // Determine the size of the input data.
31 int numRows = inputData.GetNumRows();
32 int numCols = inputData.GetNumCols();
33
34 // Create a vector for output.
35 std::vector<T> output;
36
37 // Loop through and compute means.
38 for (int j=0; j<numCols; ++j)
39 {
40 T cumulativeSum = static_cast<T>(0.0);
41 for (int i=0; i<numRows; ++i)
42 cumulativeSum += inputData.GetElement(i,j);
43
44 output.push_back(cumulativeSum / static_cast<T>(numRows));
45 }
46
47 return output;
48}

50// Function to subtract the column means.
51template <typename T>
52void SubtractColumnMeans(qbMatrix2<T> &inputData, std::vector<T> &columnMeans)
53{
54 // Determine the size of the input data.
55 int numRows = inputData.GetNumRows();
56 int numCols = inputData.GetNumCols();
57
58 // Loop through and subtract the means.
59 for (int j=0; j<numCols; ++j)
60 {
61 for (int i=0; i<numRows; ++i)
62 inputData.SetElement(i,j, inputData.GetElement(i,j) - columnMeans.at(j));
63 }
64}

66// Function to compute the covaraince matrix.
67template <typename T>
68qbMatrix2<T> ComputeCovariance(const qbMatrix2<T> &X)
69{
70 /* Compute the covariance matrix.
71 Note that here we use X'X, rather than XX' as is the usual case.
72 This is because we are requiring our data to be arranged with one
73 column (p) for each variable, with one row (k) for each observation. If
74 we computed XX', the result would be a [k x k] matrix. The covariance
75 matrix should be [p x p], so we need to transpose, hence the use of
76 X'X. */
77 int numRows = X.GetNumRows();
78 qbMatrix2<T> covX = (static_cast<T>(1.0) / static_cast<T>(numRows - 1)) * (X.Transpose() * X);
79 return covX;
80}

82// Function to compute the eigenvectors of the covariance matrix.
83template <typename T>
84int ComputeEigenvectors(const qbMatrix2<T> &covarianceMatrix, qbMatrix2<T> &eigenvectors)
85{
86 // Copy the input matrix.
87 qbMatrix2<T> X = covarianceMatrix;
88
89 // The covariance matrix must be square and symmetric.
90 if (!X.IsSquare())
91 return QBPCA_MATRIXNOTSQUARE;
92
93 // Verify that the matrix is symmetric.
94 if (!X.IsSymmetric())
95 return QBPCA_MATRIXNOTSYMMETRIC;
96
97 // Compute the eignvalues.
98 std::vector<T> eigenValues;
99 int returnStatus = qbEigQR(X, eigenValues);
100
101 // Sort the eigenvalues.
102 std::sort(eigenValues.begin(), eigenValues.end());
103 std::reverse(eigenValues.begin(), eigenValues.end());
104
105 // Compute the eigenvector for each eigenvalue.
106 qbVector<T> eV(X.GetNumCols());
107 qbMatrix2<T> eVM(X.GetNumRows(), X.GetNumCols());
108 for (int j=0; j<eigenValues.size(); ++j)
109 {
110 T eig = eigenValues.at(j);
111 int returnStatus2 = qbInvPIt<T>(X, eig, eV);
112 for (int i=0; i<eV.GetNumDims(); ++i)
113 eVM.SetElement(i, j, eV.GetElement(i));
114 }
115
116 // Return the eigenvectors.
117 eigenvectors = eVM;
118
119 // Return the final return status.
120 return returnStatus;
121}

123/* Function to compute the principal components of the supplied data. */
124template <typename T>
125int qbPCA(const qbMatrix2<T> &inputData, qbMatrix2<T> &outputComponents)
126{
127 // Make a copy of the input matrix.
128 qbMatrix2<T> X = inputData;
129
130 // Compute the mean of each column of X.
131 std::vector<T> columnMeans = ComputeColumnMeans(X);
132
133 // Subtract the column means from the data.
134 SubtractColumnMeans<T>(X, columnMeans);
135
136 // Compute the covariance matrix.
137 qbMatrix2<T> covX = ComputeCovariance(X);
138
139 // Compute the eigenvectors.
140 qbMatrix2<T> eigenvectors;
141 int returnStatus = ComputeEigenvectors(covX, eigenvectors);
142
143 // Return the output.
144 outputComponents = eigenvectors;
145
146 return returnStatus;
147}
