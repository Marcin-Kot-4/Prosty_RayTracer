8/* *************************************************************************************************
9
10 qbLSQ
11
12 Function to solve a system of linear equations using a least squares approach to handle systems
13 where there are more equations (observations) than unknowns. Assumes that the system is in the
14 form of y = X*beta.
15
16 *** INPUTS ***
17
18 Xin qbMatrix2<T> The matrix of independent variables (X in the above equation).
19 yin qbVector<T> The vector of dependent variables (y in the above equation).
20 result qbVector<T> The vector of unknown parameters (beta in the above equation).
21 The final solution is returned in this vector.
22
23 *** OUTPUTS ***
24
25 INT Flag indicating success or failure of the process.
26 1 Indicates success.
27 -1 indicates failure due to there being no computable inverse.
28
29 ************************************************************************************************* */

42// The qbLSQ function.
43template <typename T>
44int qbLSQ(const qbMatrix2<T> &Xin, const qbVector<T> &yin, qbVector<T> &result)
45{
46 // Firstly, make a copy of X and y.
47 qbMatrix2<T> X = Xin;
48 qbVector<T> y = yin;
49
50 // Compute the tranpose of X.
51 qbMatrix2<T> XT = X.Transpose();
52
53 // Compute XTX.
54 qbMatrix2<T> XTX = XT * X;
55
56 // Compute the inverse of this.
57 if (!XTX.Inverse())
58 {
59 // We were unable to compute the inverse.
60 return QBLSQ_NOINVERSE;
61 }
62
63 // Multiply the inverse by XT.
64 qbMatrix2<T> XTXXT = XTX * XT;
65
66 // And multiply by y to get the final result.
67 result = XTXXT * y;
68
69 return 1;
70}
