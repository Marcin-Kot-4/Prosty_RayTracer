12 Function to perform QR decomposition on a given input matrix.

42// The qbQR function.
43template <typename T>
44int qbQR(const qbMatrix2<T> &A, qbMatrix2<T> &Q, qbMatrix2<T> &R)
45{
46
47 // Make a copy of the input matrix.
48 qbMatrix2<T> inputMatrix = A;
49
50 // Verify that the input matrix is square.
51 if (!inputMatrix.IsSquare())
52 return QBQR_MATRIXNOTSQUARE;
53
54 // Determine the number of columns (and rows, since the matrix is square).
55 int numCols = inputMatrix.GetNumCols();
56
57 // Create a vector to store the P matrices for each column.
58 std::vector<qbMatrix2<T>> Plist;
59
60 // Loop through each column.
61 for (int j=0; j<(numCols-1); ++j)
62 {
63 // Create the a1 and b1 vectors.
64 // a1 is the column vector from A.
65 // b1 is the vector onto which we wish to reflect a1.
66 qbVector<T> a1 (numCols-j);
67 qbVector<T> b1 (numCols-j);
68 for (int i=j; i<numCols; ++i)
69 {
70 a1.SetElement(i-j, inputMatrix.GetElement(i,j));
71 b1.SetElement(i-j, static_cast<T>(0.0));
72 }
73 b1.SetElement(0, static_cast<T>(1.0));
74
75 // Compute the norm of the a1 vector.
76 T a1norm = a1.norm();
77
78 // Compute the sign we will use.
79 int sgn = -1;
80 if (a1.GetElement(0) < static_cast<T>(0.0))
81 sgn = 1;
82
83 // Compute the u-vector.
84 qbVector<T> u = a1 - (sgn * a1norm * b1);
85
86 // Compute the n-vector.
87 qbVector<T> n = u.Normalized();
88
89 // Convert n to a matrix so that we can transpose it.
90 qbMatrix2<T> nMat (numCols-j, 1);
91 for (int i=0; i<(numCols-j); ++i)
92 nMat.SetElement(i, 0, n.GetElement(i));
93
94 // Transpose nMat.
95 qbMatrix2<T> nMatT = nMat.Transpose();
96
97 // Create an identity matrix of the appropriate size.
98 qbMatrix2<T> I (numCols-j, numCols-j);
99 I.SetToIdentity();
100
101 // Compute Ptemp.
102 qbMatrix2<T> Ptemp = I - static_cast<T>(2.0) * nMat * nMatT;
103
104 // Form the P matrix with the original dimensions.
105 qbMatrix2<T> P (numCols, numCols);
106 P.SetToIdentity();
107 for (int row=j; row<numCols; ++row)
108 {
109 for (int col=j; col<numCols; ++col)
110 {
111 P.SetElement(row, col, Ptemp.GetElement(row-j, col-j));
112 }
113 }
114
115 // Store the result into the Plist vector.
116 Plist.push_back(P);
117
118 // Apply this transform matrix to inputMatrix and use this result
119 // next time through the loop.
120 inputMatrix = P * inputMatrix;
121 }
122
123 // Compute Q.
124 qbMatrix2<T> Qmat = Plist.at(0);
125 for (int i=1; i<(numCols-1); ++i)
126 {
127 Qmat = Qmat * Plist.at(i).Transpose();
128 }
129
130 // Return the Q matrix.
131 Q = Qmat;
132
133 // Compute R.
134 int numElements = Plist.size();
135 qbMatrix2<T> Rmat = Plist.at(numElements-1);
136 for (int i=(numElements-2); i>=0; --i)
137 {
138 Rmat = Rmat * Plist.at(i);
139 }
140 Rmat = Rmat * A;
141
142 // And return the R matrix.
143 R = Rmat;
144
145}
