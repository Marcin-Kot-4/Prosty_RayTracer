23// Function to estimate (real) eigenvalues using QR decomposition.
24/* Note that this is only valid for matrices that have ALL real
25 eigenvalues. The only matrices that are guaranteed to have only
26 real eigenvalues are symmetric matrices. Therefore, this function
27 is only guaranteed to work with symmetric matrices. */
28template <typename T>
29int qbEigQR(const qbMatrix2<T> &inputMatrix, std::vector<T> &eigenValues)
30{
31 // Make a copy of the input matrix.
32 qbMatrix2<T> A = inputMatrix;
33
34 // Verify that the input matrix is square.
35 if (!A.IsSquare())
36 return QBEIG_MATRIXNOTSQUARE;
37
38 // Verify that the matrix is symmetric.
39 if (!A.IsSymmetric())
40 return QBEIG_MATRIXNOTSYMMETRIC;
41
42 // The number of eigenvalues is equal to the number of rows.
43 int numRows = A.GetNumRows();
44
45 // Create an identity matrix of the same dimensions.
46 qbMatrix2<T> identityMatrix(numRows, numRows);
47 identityMatrix.SetToIdentity();
48
49 // Create matrices to store Q and R.
50 qbMatrix2<T> Q (numRows, numRows);
51 qbMatrix2<T> R (numRows, numRows);
52
53 // Loop through each iteration.
54 int maxIterations = 10e3;
55 int iterationCount = 0;
56 bool continueFlag = true;
57 while ((iterationCount < maxIterations) && continueFlag)
58 {
59 // Compute the QR decomposition of A.
60 int returnValue = qbQR<T>(A, Q, R);
61
62 // Compute the next value of A as the product of R and Q.
63 A = R * Q;
64
65 /* Check if A is now close enough to being upper-triangular.
66 We can do this using the IsRowEchelon() function from the
67 qbMatrix2 class. */
68 if (A.IsRowEchelon())
69 continueFlag = false;
70
71 // Increment iterationCount.
72 iterationCount++;
73 }
74
75 // At this point, the eigenvalues should be the diagonal elements of A.
76 for (int i=0; i<numRows; ++i)
77 eigenValues.push_back(A.GetElement(i,i));
78
79 // Set the return status accordingly.
80 if (iterationCount == maxIterations)
81 return QBEIG_MAXITERATIONSEXCEEDED;
82 else
83 return 0;
84
85}

87// Function to perform inverse power iteration method.
88template <typename T>
89int qbInvPIt(const qbMatrix2<T> &inputMatrix, const T &eigenValue, qbVector<T> &eigenVector)
90{
91 // Make a copy of the input matrix.
92 qbMatrix2<T> A = inputMatrix;
93
94 // Verify that the input matrix is square.
95 if (!A.IsSquare())
96 return QBEIG_MATRIXNOTSQUARE;
97
98 // Setup a random number generator.
99 std::random_device myRandomDevice;
100 std::mt19937 myRandomGenerator(myRandomDevice());
101 std::uniform_int_distribution<int> myDistribution(1.0, 10.0);
102
103 /* The number of eigenvectors and eigenvalues that we will compute will be
104 equal to the number of rows in the input matrix. */
105 int numRows = A.GetNumRows();
106
107 // Create an identity matrix of the same dimensions.
108 qbMatrix2<T> identityMatrix(numRows, numRows);
109 identityMatrix.SetToIdentity();
110
111 // Create an initial vector, v.
112 qbVector<T> v(numRows);
113 for (int i=0; i<numRows; ++i)
114 v.SetElement(i, static_cast<T>(myDistribution(myRandomGenerator)));
115
116 // Iterate.
117 int maxIterations = 100;
118 int iterationCount = 0;
119 T deltaThreshold = static_cast<T>(1e-9);
120 T delta = static_cast<T>(1e6);
121 qbVector<T> prevVector(numRows);
122 qbMatrix2<T> tempMatrix(numRows, numRows);
123
124 while ((iterationCount < maxIterations) && (delta > deltaThreshold))
125 {
126 // Store a copy of the current working vector to use for computing delta.
127 prevVector = v;
128
129 // Compute the next value of v.
130 tempMatrix = A - (eigenValue * identityMatrix);
131 tempMatrix.Inverse();
132 v = tempMatrix * v;
133 v.Normalize();
134
135 // Compute delta.
136 delta = (v - prevVector).norm();
137
138 // Increment iteration count.
139 iterationCount++;
140 }
141
142 // Return the estimated eigenvector.
143 eigenVector = v;
144
145 // Set the return status accordingly.
146 if (iterationCount == maxIterations)
147 return QBEIG_MAXITERATIONSEXCEEDED;
148 else
149 return 0;
150
151}

153// The qbEIG function (power iteration method).
154template <typename T>
155int qbEIG_PIt(const qbMatrix2<T> &X, T &eigenValue, qbVector<T> &eigenVector)
156{
157 // Make a copy of the input matrix.
158 qbMatrix2<T> inputMatrix = X;
159
160 // Verify that the input matrix is square.
161 if (!inputMatrix.IsSquare())
162 return QBEIG_MATRIXNOTSQUARE;
163
164 // Setup a random number generator.
165 std::random_device myRandomDevice;
166 std::mt19937 myRandomGenerator(myRandomDevice());
167 std::uniform_int_distribution<int> myDistribution(1.0, 10.0);
168
169 /* The number of eigenvectors and eigenvalues that we will compute will be
170 equal to the number of rows in the input matrix. */
171 int numRows = inputMatrix.GetNumRows();
172
173 // Create an identity matrix of the same dimensions.
174 qbMatrix2<T> identityMatrix(numRows, numRows);
175 identityMatrix.SetToIdentity();
176
177 /* **************************************************************
178 Compute the eigenvector.
179 ************************************************************** */
180
181 // Create an initial vector, v.
182 qbVector<T> v(numRows);
183 for (int i=0; i<numRows; ++i)
184 v.SetElement(i, static_cast<T>(myDistribution(myRandomGenerator)));
185
186 // Loop over the required number of iterations.
187 qbVector<T> v1(numRows);
188 int numIterations = 1000;
189 for (int i=0; i<numIterations; ++i)
190 {
191 v1 = inputMatrix * v;
192 v1.Normalize();
193 v = v1;
194 }
195
196 // Store this eigenvector.
197 eigenVector = v1;
198
199 /* **************************************************************
200 Compute the eigenvalue corresponding to this eigenvector.
201 ************************************************************** */
202
203 // Compute the cumulative sum.
204 T cumulativeSum = static_cast<T>(0.0);
205 for (int i=1; i<numRows; ++i)
206 cumulativeSum += inputMatrix.GetElement(0,i) * v1.GetElement(i);
207
208 eigenValue = (cumulativeSum / v1.GetElement(0)) + inputMatrix.GetElement(0,0);
209
210 return 0;
211}
